# Nginx Configuration for aadikarta.org

# 1. Main Website (aadikarta.org) - Currently "Coming Soon"
server {
    server_name aadikarta.org www.aadikarta.org;

    # Redirect to HTTPS (once SSL is configured)
    # return 301 https://$host$request_uri;

    location / {
        proxy_pass http://127.0.0.1:3002; # Points to the same Astro/React app
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        
        # Ensure we always show the coming soon page on the main domain
        # The app logic in App.tsx will handle the display based on hostname
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/aadikarta.org/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/aadikarta.org/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot




}

# 3. API Service (api.aadikarta.org)
server {
    server_name api.aadikarta.org;

    location / {
        proxy_pass http://127.0.0.1:8000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }

    # WebSocket for Chat
    location /chat/ws {
        proxy_pass http://127.0.0.1:8000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";
        proxy_set_header Host $host;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/aadikarta.org/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/aadikarta.org/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot


}

# 4. Admin Dashboard (admin.aadikarta.org)
server {
    server_name admin.aadikarta.org;

    location / {
        proxy_pass http://127.0.0.1:3001;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/aadikarta.org/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/aadikarta.org/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot


}
